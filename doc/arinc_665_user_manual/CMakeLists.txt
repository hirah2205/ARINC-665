cmake_minimum_required( VERSION 3.31 )

find_program( ASCIIDOCTOR_EXE NAMES asciidoctor DOC "Ascii Doctor executable" )

if( NOT ASCIIDOCTOR_EXE )
  return()
endif()

add_custom_target(
  arinc_665_user_manual
  COMMAND
    ${ASCIIDOCTOR_EXE}
    --backend html5
    --source-dir ${CMAKE_CURRENT_SOURCE_DIR}
    --destination-dir ${CMAKE_CURRENT_BINARY_DIR}
    ${CMAKE_CURRENT_SOURCE_DIR}/arinc_665_user_manual.adoc
  COMMENT "Build arinc_665_user_manual.html" )

add_dependencies( doc arinc_665_user_manual )

install(
  FILES
    ${CMAKE_CURRENT_BINARY_DIR}/arinc_665_user_manual.html
  TYPE DOC
  COMPONENT runtime
  OPTIONAL )
